# backend/websocket/connection_manager.py

```python
from fastapi import WebSocket


class ConnectionManager:

    def __init__(self):
        self.connections = {}

    async def connect(self, execution_id: str, websocket: WebSocket):

        await websocket.accept()
        self.connections[execution_id] = websocket

    def disconnect(self, execution_id: str):

        if execution_id in self.connections:
            del self.connections[execution_id]

    async def send_output(self, execution_id: str, message: str):

        websocket = self.connections.get(execution_id)

        if websocket:
            await websocket.send_text(message)
```

# backend/executor/execution_manager.py

```python
import threading
from backend.executor.code_runner import CodeRunner
from backend.utils.id_generator import generate_execution_id


class ExecutionManager:

    def __init__(self):
        self.status = {}

    def submit_execution(self, language: str, code: str, stdin: str):

        execution_id = generate_execution_id()

        self.status[execution_id] = "running"

        thread = threading.Thread(
            target=self._run_execution,
            args=(execution_id, language, code, stdin)
        )

        thread.start()

        return execution_id

    def _run_execution(self, execution_id, language, code, stdin):

        runner = CodeRunner()

        try:
            result = runner.run(language, code, stdin)

            self.status[execution_id] = "completed"

        except Exception:

            self.status[execution_id] = "failed"

    def get_status(self, execution_id):

        return {
            "execution_id": execution_id,
            "status": self.status.get(execution_id, "unknown")
        }
```

# backend/executor/code_runner.py

```python
import subprocess
import tempfile
from pathlib import Path


class CodeRunner:

    def run(self, language: str, code: str, stdin: str):

        if language == "python":
            return self._run_python(code, stdin)

        raise ValueError("Unsupported language")

    def _run_python(self, code: str, stdin: str):

        with tempfile.TemporaryDirectory() as tmp:

            file_path = Path(tmp) / "main.py"

            with open(file_path, "w") as f:
                f.write(code)

            process = subprocess.run(
                ["python", str(file_path)],
                input=stdin,
                capture_output=True,
                text=True,
                timeout=5
            )

            return process.stdout
```

import threading


class TimeoutGuard:

    def __init__(self, timeout_seconds: int):
        self.timeout = timeout_seconds

    def run_with_timeout(self, func, *args):

        thread = threading.Thread(target=func, args=args)
        thread.start()

        thread.join(self.timeout)

        if thread.is_alive():
            raise TimeoutError("Execution timed out")

class WorkerManager:

# `sandbox/policies/seccomp_profile.json`

This **restricts system calls** allowed inside the container.

```json
{
  "defaultAction": "SCMP_ACT_ERRNO",
  "architectures": [
    "SCMP_ARCH_X86_64"
  ],
  "syscalls": [
    {
      "names": [
        "read",
        "write",
        "exit",
        "exit_group",
        "brk",
        "mmap",
        "munmap",
        "fstat",
        "close"
      ],
      "action": "SCMP_ACT_ALLOW"
    }
  ]
}
```

---

# `sandbox/policies/resource_limits.json`

Defines **execution limits** to prevent abuse.

```json
{
  "cpu_limit": "1",
  "memory_limit_mb": 256,
  "max_execution_time_sec": 5,
  "max_processes": 20,
  "max_file_size_kb": 1024
}
```

    const response = await fetch("/api/execute", {

# `frontend/js/websocket.js`

```javascript id="websocket_js"
let ws;

function initWebSocket(executionId) {
    ws = createWebSocket(`ws://${window.location.host}/ws/${executionId}`);

    ws.onmessage = (event) => {
        appendToTerminal(event.data);
    };

    ws.onclose = () => {
        appendToTerminal("Execution finished.\n");
    };
}
```

Here’s a `docker-compose.yml` for your **browser-based code editor and executor** project, integrating **API server, worker, and sandbox execution environment**:

```yaml
version: "3.9"

services:
  api:
    container_name: code_runner_api
    build:
      context: .
      dockerfile: ./backend/Dockerfile.api
    ports:
      - "8000:8000"
    environment:
      - ENV=development
    volumes:
      - ./backend:/app/backend
      - ./logs:/app/logs
      - ./sandbox:/app/sandbox
    depends_on:
      - worker

  worker:
    container_name: code_runner_worker
    build:
      context: .
      dockerfile: ./backend/Dockerfile.worker
    environment:
      - ENV=development
    volumes:
      - ./backend:/app/backend
      - ./logs:/app/logs
      - ./sandbox:/app/sandbox
    depends_on:
      - queue

  queue:
    container_name: code_runner_queue
    image: redis:7
    ports:
      - "6379:6379"

  sandbox_python:
    container_name: sandbox_python
    build:
      context: ./sandbox/docker
      dockerfile: Dockerfile.python
    stdin_open: true
    tty: true
    volumes:
      - ./sandbox/runtime:/sandbox/runtime
    networks:
      - sandbox_net

  sandbox_cpp:
    container_name: sandbox_cpp
    build:
      context: ./sandbox/docker
      dockerfile: Dockerfile.cpp
    stdin_open: true
    tty: true
    volumes:
      - ./sandbox/runtime:/sandbox/runtime
    networks:
      - sandbox_net

  sandbox_node:
    container_name: sandbox_node
    build:
      context: ./sandbox/docker
      dockerfile: Dockerfile.node
    stdin_open: true
    tty: true
    volumes:
      - ./sandbox/runtime:/sandbox/runtime
    networks:
      - sandbox_net

networks:
  sandbox_net:
    driver: bridge
```

### ✅ Explanation:

1. **`api` service**

   * Hosts the backend API (`Flask`/`FastAPI` etc.)
   * Maps port 8000 to host
   * Mounts backend, logs, and sandbox folders

2. **`worker` service**

   * Processes code execution jobs asynchronously
   * Reads from `queue` (Redis)

3. **`queue` service**

   * Uses **Redis** as a job queue

4. **`sandbox_*` services**

   * Each language runs in a separate Docker container
   * Isolated execution environment with mounted runtime scripts

5. **`networks`**

   * A bridge network (`sandbox_net`) ensures sandboxes can communicate if needed
